# 判断空值方法

```tsx
// 根据Data是否存在值，使用some（）方法判断每一个值是否为空
if (Object.values(data).some((value) => value === ""))
  return message.error("请输入内容，相应内容不能为空");
```

```tsx
if (
  Object.keys(data) // 返回一个由属性名组成的数组
    .filter((key) => key !== "xxx") // 移除数组中属性名为 "xxx" 的元素，生成新的数组
    .some((key) => data[key as keyof IProps] === "")
)
  return message.error("请输入内容，相应内容不能为空");
```

```tsx
// 按逻辑判断字段格式
const judgment = (data: IProps) => {
  if (data.start === "") {
    message.info("开始时间未填写");

    return false;
  } else if (data.end === "") {
    message.info("结束时间未填写");

    return false;
  } else if (data.start > data.end) {
    message.info("开始时间不能大于结束时间");

    return false;
  } else if (data.backToWork === "") {
    message.info("回到工作时间未填写");

    return false;
  } else if (data.uploadFile === null) {
    message.info("附件未添加");

    return false;
  } else {
    return true;
  }
};
```
